clear; 
clc; 
x = input ("enter the value of x="); 
h = input ("enter the value of h="); 
subplot(3,1,1); 
plot2d3(x); 
xlabel("time index n"); 
ylabel("amplitude"); 
title ("sequence of x(n)"); 
subplot(3,1,2); 
plot2d3(h); 
xlabel("time index n"); 
ylabel("amplitude"); 
title("sequence of h(n)"); 
b=xcorr(x,h); 
disp(b); 
subplot(3,1,3); 
plot2d3(b); 
xlabel("time index n"); 
ylabel("amplitude"); 
title("cross correlation");
