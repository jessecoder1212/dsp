#include <stdio.h>
#include <math.h>

#define PI 3.1416

float x[256], y[256];

int main()
{
    float fs = 10000.0;
    float T = 1.0 / fs;
    float t = 0.0;

    int i;

    // Generate input signal
    for(i = 0; i < 256; i++, t += T)
    {
        x[i] = sin(2 * PI * 117.18 * t)
             + sin(2 * PI * 585.93 * t)
             + sin(2 * PI * 2226.25 * t);
    }

    // FIR filter using Hanning window coefficients
    for(i = 6; i < 256; i++)
    {
        y[i] =
              0.0000 * x[i]
            + 0.0833 * x[i - 1]
            + 0.2500 * x[i - 2]
            + 0.3333 * x[i - 3]
            + 0.2500 * x[i - 4]
            + 0.0833 * x[i - 5]
            + 0.0000 * x[i - 6];
    }

    return 0;
}
