#include <stdio.h>
#include <math.h>

#define PI 3.14159264

float x[256], y[256];

int main()
{
    float fs = 10000.0;
    float T = 1.0 / fs;
    float t = 0.0;

    int i;

    // Generate input signal
    for(i = 0; i < 256; i++)
    {
        x[i] = sin(2 * PI * 117.18 * t)
             + sin(2 * PI * 585.93 * t)
             + sin(2 * PI * 3906.25 * t);

        t += T;
    }

    // Initial conditions
    y[0] = 0;
    y[1] = 0;

    // Second order Butterworth LPF
    for(i = 2; i < 256; i++)
    {
        y[i] = -0.3695274 * y[i-1]
             - 0.1958157 * y[i-2]
             + 0.3913358 * (x[i] + x[i-2])
             + 0.7826715 * x[i-1];
    }

    return 0;
}
